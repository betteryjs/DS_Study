### 实现打印一个整数的二进制

```cpp
string pringInt( unsigned int & num){
    string  res{};
    for (int i = 31; i >=0 ; --i) {
        res+=(num&(1<<i))==0 ? "0": "1";
    }
    return res;
}
```

### 快速幂
```cpp

/*
现在有 N 个格子排成一排 ， 假设你有 M 种的颜色
现在要你给这N个格子都涂上颜色，且每相邻的两个格子的颜色都不能相同，现在要你求有多少种方法涂色方案
由于答案太大，每次结果对 1000000007 取模
*/

#define ll long long
#define mod 1000000007

ll fast_Pow(ll a,ll b){
    ll ans = 1;
    ll base = a;
    while(b){
        if(b & 1)
            ans = ans * base % mod;
        base = base * base % mod;
        b >>= 1;
    }
    return ans;
}
```
